Simple Integration

GK-Chart is real easy to integrate with html and Js. See below exaple and fell yourself the easiyness of Gk-Chart.

  • HTML
  • JavaScript 
&lt;!DOCTYPE html&gt;
&lt;html lang=&quot;en&quot;&gt;
  &lt;head&gt;
    &lt;meta charset=&quot;UTF-8&quot;&gt;
    &lt;meta name=&quot;viewport&quot; content=&quot;width=device-width, initial-scale=1.0&quot;&gt;
    &lt;meta http-equiv=&quot;X-UA-Compatible&quot; content=&quot;ie=edge&quot;&gt;
    &lt;title&gt;Document&lt;/title&gt;
  &lt;/head&gt;
  &lt;body&gt;
    &lt;div id=&quot;chartSmoothLine&quot; style=&quot;height:350px; width:100%;&quot;&gt;&lt;/div&gt;
    &lt;script type=&quot;text/javascript&quot; src=&quot;gk-chart.min.js&quot;&gt;&lt;/script&gt;
  &lt;/body&gt;
&lt;/html&gt;

let datapoints = [
  { label: "January", y: 40 },
  { label: "Feburary", y: 180 },
  { label: "March", y: 60 },
  { label: "April", y: 300 },
  { label: "May", y: 140 },
  { label: "June", y: 240 }
];

var chartSmoothLine = {
  config: {
    title: "Line Chart",
    chartType: "line",
    printEnable: true
  },
  data: [
    {
      chartColor: "#00d554",
      fill: false,
      dataLabel: "Data Set 1",
      datapoints: datapoints
    }
  ]
};

new GKChart({
  id: "chartSmoothLine",
  data: chartSmoothLine
}) ;
